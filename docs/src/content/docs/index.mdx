import { Card, CardGrid } from '@astrojs/starlight/components';

<div class="hero-intro">
  Pulse differs from traditional signals or state managers by treating **Conditions** as first-class citizens. Instead of embedding complex boolean logic inside components, you define **Semantic Guards** that can be observed, composed, and debugged independently.
</div>

## Core Primitives

<CardGrid>
	<Card title="Pulse Objects" icon="puzzle">
        **Universal Proxy Reactivity**
        `pulse()` creates deep reactive proxies that track property access and mutation automatically.
        
        ```ts
        const state = pulse({ count: 0 });
        state.count++; // It's reactive!
        ```
	</Card>
	<Card title="Semantic Guards" icon="approve-check-circle">
        **First-Class Conditions**
		Guards serve as high-performance **Selectors** with built-in dependency tracking and failing context.
        
        ```ts
        const isAdmin = guard("is-admin", () => {
           return user.role === "admin"; 
        });
        ```
	</Card>
    <Card title="Logic Composition" icon="list-format">
        **Declarative Combinators**
        Combine guards using built-in logical helpers like `all`, `any`, and `not`.

        ```ts
        const canLaunch = guard.all([
            isReady,
            isUserAuthorized,
            guard.not(isMaintenance)
        ]);
        ```
    </Card>
</CardGrid>

## Pro Features

<CardGrid>
	<Card title="Async & Race Control" icon="star">
		**Native Async Support**
        Guards handle promises natively. Pulse implements internal versioning to automatically cancel stale evaluations.

        ```ts
        const data = guard("fetch", async () => {
             return (await fetch("/api")).json();
        });
        ```
	</Card>
    <Card title="Explainable Logic" icon="magnifier">
		**Deep Guard Inspection**
        Get a structured tree of the current status, recursive dependency failures, and failure reason codes.

        ```ts
        const explanation = canCheckout.explain();
        // { status: 'fail', reason: { code: 'AUTH' }, ... }
        ```
	</Card>
    <Card title="Isomorphic SSR" icon="laptop">
		**Astro & React Ready**
        Evaluate guards on the server during SSR and have them seamlessly hydrate on the client.
	</Card>
</CardGrid>
